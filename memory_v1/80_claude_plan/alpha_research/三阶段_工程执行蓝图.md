```python
"""
从现有Parquet读取数据，格式化为alphalens所需格式
"""
import pandas as pd
from pathlib import Path

PARQUET_DIR = Path("runtime/market_data")

def load_prices(instruments: list[str], start: str, end: str) -> pd.DataFrame:
    """
    返回：prices DataFrame
      index: DatetimeIndex（交易日）
      columns: instrument_code
      values: 后复权收盘价
    """
    dfs = []
    for code in instruments:
        path = PARQUET_DIR / f"{code}.parquet"
        df = pd.read_parquet(path)
        df = df.loc[start:end, ['close']].rename(columns={'close': code})
        dfs.append(df)
    return pd.concat(dfs, axis=1)


def load_factor_input(instruments: list[str], start: str, end: str) -> pd.DataFrame:
    """
    返回：raw_df DataFrame（含计算因子所需的所有列）
      index: DatetimeIndex
      columns: instrument_code × (open/high/low/close/volume)
    用MultiIndex DataFrame方便批量计算
    """
    dfs = {}
    for code in instruments:
        path = PARQUET_DIR / f"{code}.parquet"
        df = pd.read_parquet(path).loc[start:end]
        dfs[code] = df
    return pd.concat(dfs, axis=1, keys=dfs.keys())
```

```python
"""
Phase 1 基础因子：动量 / 换手率 / 市净率
输出格式：pd.Series with MultiIndex(date, asset)，这是alphalens要求的格式
"""
import pandas as pd
import numpy as np
import akshare as ak


def factor_momentum_20d(prices: pd.DataFrame) -> pd.Series:
    """
    20日动量：T日因子值 = T-1日收盘 / T-21日收盘 - 1
    时间对齐：用T-1日数据，避免用当日收盘产生前向偏差
    """
    ret = prices.pct_change(20).shift(1)  # shift(1)：T日用T-1日的动量值
    # 转换为alphalens要求的MultiIndex格式
    factor = ret.stack()
    factor.index.names = ['date', 'asset']
    return factor.dropna()


def factor_turnover_20d(instruments: list[str], start: str, end: str) -> pd.Series:
    """
    20日平均换手率：从AkShare获取，shift(1)对齐
    """
    dfs = []
    for code in instruments:
        try:
            df = ak.stock_zh_a_hist(
                symbol=code.split('.')[0],
                period="daily",
                start_date=start.replace('-', ''),
                end_date=end.replace('-', ''),
                adjust="hfq"
            )[['日期', '换手率']].rename(columns={'日期': 'date', '换手率': 'turnover'})
            df['date'] = pd.to_datetime(df['date'])
            df = df.set_index('date')
            df['asset'] = code
            df['factor'] = df['turnover'].rolling(20).mean().shift(1)
            dfs.append(df[['asset', 'factor']].dropna())
        except Exception:
            continue

    result = pd.concat(dfs)
    result = result.reset_index().set_index(['date', 'asset'])['factor']
    return result


def factor_pb_ratio(instruments: list[str], start: str, end: str) -> pd.Series:
    """
    市净率（PB）：从AkShare获取，低PB = 高因子分（取倒数）
    注意：财务数据以季度频率更新，用最近公告的数据前向填充
    """
    dfs = []
    for code in instruments:
        try:
            df = ak.stock_financial_analysis_indicator(
                symbol=code.split('.')[0]
            )[['报告期', '每股净资产']].rename(
                columns={'报告期': 'report_date', '每股净资产': 'nav_per_share'}
            )
            # TODO: 这里需要用公告日而非报告期，Phase 1先用报告期近似，Phase 2修正
            # ⚠️ 这是已知的简化，务必在Phase 2修复前向偏差
            df['report_date'] = pd.to_datetime(df['report_date'])
            df = df.set_index('report_date').sort_index()
            dfs.append((code, df))
        except Exception:
            continue

    # 此处省略完整的价格/净资产对齐逻辑，执行时补全
    # 返回格式：MultiIndex(date, asset) → pb_inverse（1/PB，高值=低估值）
    raise NotImplementedError("PB因子数据对齐逻辑待Phase 1执行时补全")
```

### 任务2.1：IC批量评估流水线

**文件**：`src/alpha_research/evaluation/ic_pipeline.py`

```python
"""
批量评估候选因子的IC/ICIR/衰减
"""
from joblib import Parallel, delayed
import alphalens as al
import pandas as pd

def evaluate_single_factor(
    factor_series: pd.Series,
    prices: pd.DataFrame,
    factor_name: str
) -> dict:
    """对单个因子跑完整IC评估，返回结果字典"""
    try:
        factor_data = al.utils.get_clean_factor_and_forward_returns(
            factor=factor_series,
            prices=prices,
            periods=(1, 5, 10, 20),
            quantiles=5,
            max_loss=0.30
        )
        ic = al.performance.factor_information_coefficient(factor_data)
        icir = ic.mean() / ic.std()

        # 衰减半衰期：找IC从最高点降到一半的周期数
        ic_by_period = ic.mean()
        max_ic = ic_by_period.abs().max()
        decay_halflife = None
        prev = None
        for period, val in ic_by_period.abs().items():
            if prev is not None and val <= max_ic / 2:
                decay_halflife = period
                break
            prev = val

        return {
            "factor_name": factor_name,
            "ic_mean": ic_by_period.to_dict(),
            "icir": icir.to_dict(),
            "decay_halflife_days": decay_halflife,
            "n_obs": len(factor_data),
            "error": None
        }
    except Exception as e:
        return {"factor_name": factor_name, "error": str(e)}


def batch_evaluate(factor_dict: dict, prices: pd.DataFrame, n_jobs: int = 4) -> list[dict]:
    """
    factor_dict: {factor_name: factor_series}
    并行评估所有因子
    """
    results = Parallel(n_jobs=n_jobs)(
        delayed(evaluate_single_factor)(series, prices, name)
        for name, series in factor_dict.items()
    )
    return results
```

```python
"""
因子筛选规则（三关）
"""

def screen_factor(ic_result: dict) -> tuple[bool, str]:
    """
    Returns: (passed, reason)
    """
    # 关1：IC均值绝对值（用10日周期作为主要参考）
    ic_10d = abs(ic_result.get("ic_mean", {}).get(10, 0))
    if ic_10d < 0.03:
        return False, f"IC均值不足：{ic_10d:.4f} < 0.03"

    # 关2：ICIR（稳定性）
    icir_10d = abs(ic_result.get("icir", {}).get(10, 0))
    if icir_10d < 0.5:
        return False, f"ICIR不足：{icir_10d:.4f} < 0.5"

    # 关3：衰减半衰期
    halflife = ic_result.get("decay_halflife_days")
    if halflife is None or halflife < 5:
        return False, f"衰减过快：半衰期={halflife}日 < 5日"

    return True, "通过全部筛选条件"
```

### 任务2.3：相关性去重

**文件**：`src/alpha_research/evaluation/correlation.py`

```python
"""
对通过筛选的因子做相关性去重
相关性 > 0.7 的因子对，保留ICIR更高的
"""
import pandas as pd
import numpy as np
from scipy.stats import spearmanr

def deduplicate_factors(
    factor_dict: dict,       # {name: series}
    ic_results: dict,        # {name: ic_result}
    corr_threshold: float = 0.7
) -> list[str]:
    """返回去重后的因子名称列表"""
    names = list(factor_dict.keys())
    n = len(names)

    # 计算两两Spearman相关系数（截面Rank相关）
    corr_matrix = pd.DataFrame(index=names, columns=names, dtype=float)
    for i in range(n):
        for j in range(i, n):
            common_idx = factor_dict[names[i]].index.intersection(
                factor_dict[names[j]].index
            )
            if len(common_idx) < 100:
                corr = 0.0
            else:
                corr, _ = spearmanr(
                    factor_dict[names[i]].loc[common_idx],
                    factor_dict[names[j]].loc[common_idx]
                )
            corr_matrix.loc[names[i], names[j]] = abs(corr)
            corr_matrix.loc[names[j], names[i]] = abs(corr)

    # 贪心去重：每次移除ICIR更低的那个
    kept = list(names)
    removed = set()
    for i in range(n):
        if names[i] in removed:
            continue
        for j in range(i + 1, n):
            if names[j] in removed:
                continue
            if corr_matrix.loc[names[i], names[j]] > corr_threshold:
                icir_i = abs(ic_results[names[i]].get("icir", {}).get(10, 0))
                icir_j = abs(ic_results[names[j]].get("icir", {}).get(10, 0))
                if icir_i >= icir_j:
                    removed.add(names[j])
                else:
                    removed.add(names[i])
                    break

    return [n for n in kept if n not in removed]
```

### 任务3.1：复合信号生成

**文件**：`src/alpha_research/signal_composer.py`

```python
"""
将多个因子组合为截面Rank评分
输出：每个交易日每个标的的复合信号分数
"""
import pandas as pd
import numpy as np

def compose_signal(
    factor_dict: dict,    # {factor_name: pd.Series with MultiIndex(date, asset)}
    weights: dict         # {factor_name: float}，权重之和不必为1（会normalize）
) -> pd.Series:
    """
    Returns: composite_score，MultiIndex(date, asset)
    值越高 = 买入倾向越强
    """
    # 截面Rank：每个日期内标准化为0-1之间的排名
    ranked = {}
    for name, series in factor_dict.items():
        ranked[name] = series.groupby(level='date').rank(pct=True)

    # 加权求和
    total_weight = sum(weights.values())
    composite = sum(
        ranked[name] * (weights[name] / total_weight)
        for name in weights
        if name in ranked
    )
    return composite


def generate_top_n_signal(
    composite_score: pd.Series,
    top_pct: float = 0.1    # 取top 10%作为买入候选
) -> pd.Series:
    """返回0/1信号：1=买入候选，0=不操作"""
    threshold = composite_score.groupby(level='date').quantile(1 - top_pct)
    signal = composite_score >= composite_score.index.get_level_values('date').map(threshold)
    return signal.astype(int)
```
